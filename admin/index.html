أنشئ جداول قاعدة البيانات

انسخ هذا الكود وشغّله في SQL Editor:

-- جدول المحتوى الرئيسي CREATE TABLE IF NOT EXISTS aura_content ( id bigserial primary key, section text unique not null, data jsonb not null default '{}', updated_at timestamptz default now() ); -- جدول الحجوزات CREATE TABLE IF NOT EXISTS aura_bookings ( id uuid default gen_random_uuid() primary key, name text, email text, date text, time text, service text, message text, status text default 'pending', created_at timestamptz default now() ); -- ═══ صلاحيات آمنة (RLS) ═══ ALTER TABLE aura_content ENABLE ROW LEVEL SECURITY; ALTER TABLE aura_bookings ENABLE ROW LEVEL SECURITY; -- المحتوى: الزوار يقرأون فقط / المسؤول (مسجّل دخول) يعدّل DROP POLICY IF EXISTS "content_read" ON aura_content; DROP POLICY IF EXISTS "content_write" ON aura_content; CREATE POLICY "content_read" ON aura_content FOR SELECT USING (true); CREATE POLICY "content_write" ON aura_content FOR ALL TO authenticated USING (true) WITH CHECK (true); -- الحجوزات: الزوار يرسلون فقط / المسؤول وحده يقرأ ويعدّل DROP POLICY IF EXISTS "booking_insert" ON aura_bookings; DROP POLICY IF EXISTS "booking_admin" ON aura_bookings; CREATE POLICY "booking_insert" ON aura_bookings FOR INSERT WITH CHECK (true); CREATE POLICY "booking_admin" ON aura_bookings FOR ALL TO authenticated USING (true) WITH CHECK (true);

🔒 هذه الصلاحيات آمنة: الزوار يقرأون المحتوى ويرسلون الحجوزات فقط — لكن لا أحد يستطيع تعديل موقعك أو قراءة بيانات عملائك إلا أنت بعد تسجيل الدخول.
